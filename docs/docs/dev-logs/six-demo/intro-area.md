```js component
import Component from '@glimmer/component';
import { tracked, cached } from '@glimmer/tracking';
import { task } from 'ember-concurrency';
import { flatGroup } from 'd3-array';

export default class extends Component {
  @tracked data = [];

  load = task(async () => {
    const req = await fetch('/datasets/cars.json');
    this.data = await req.json();
  });

  constructor(owner, args) {
    super(owner, args);
    this.load.perform();
  }

  @cached get carsByCylinders() {
    const grouped = flatGroup(this.data, d => d.Cylinders, d => d.Origin);
    return grouped.map(([cylinders, region, cars]) => ({
      cylinders,
      region,
      cars: cars.length,
    })).sortBy('cylinders');
  }

  @cached get cylinders() {
    return Array.from(new Set(this.carsByCylinders.map(d => d.cylinders)));
  }
}
```
